Model

Can:

  • Ember.set('property', value); //update properties
  • NB.EventHub.dispatch('PROPERTY.CHANGED', value); //emit events
  • $.Deferred().resolve(value); //resolve signals

Cannot:

  • $('.property').hide();//manipulate directly DOM elements
  • this.controller.XXX; //access controller
  • this.view.XXX; //access view
